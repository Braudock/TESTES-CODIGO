%macro load_and_tag_datasets(libname);

    %do i = 1 %to 12;
        %let month = %sysfunc(putn(&i, z2.));
        %let dataset_name = E2023&month;

        %if %sysfunc(exist(&libname..&dataset_name)) %then %do;
            data &libname..&dataset_name;
                set &libname..&dataset_name;
                ANOMES = "2023&month";
            run;
        %end;
    %end;

%mend load_and_tag_datasets;

%load_and_tag_datasets(libname=YourLibName);
