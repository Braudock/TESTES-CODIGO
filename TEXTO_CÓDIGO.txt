import pandas as pd

# Lista dos caminhos dos arquivos Excel
caminhos = [
    "caminho/para/o/arquivo1.xlsx",
    "caminho/para/o/arquivo2.xlsx",
    # Adicione mais caminhos conforme necessário
]

# Dicionário para armazenar o nome do arquivo e a quantidade de linhas
contagem_linhas = {}

for caminho in caminhos:
    try:
        # Lendo o arquivo Excel
        df = pd.read_excel(caminho)
        # Contando as linhas e armazenando no dicionário
        contagem_linhas[caminho] = len(df)
    except Exception as e:
        # Em caso de erro, armazena a mensagem de erro
        contagem_linhas[caminho] = str(e)

# Exibindo os resultados
for arquivo, linhas in contagem_linhas.items():
    print(f"Arquivo: {arquivo}, Linhas: {linhas}")
