proc sql;
   update tabela_original
   set campo1 = (select campo1 from tabela_atualizacao where tabela_original.chave = tabela_atualizacao.chave),
       campo2 = (select campo2 from tabela_atualizacao where tabela_original.chave = tabela_atualizacao.chave)
   where exists (select * from tabela_atualizacao where tabela_original.chave = tabela_atualizacao.chave);
quit;
