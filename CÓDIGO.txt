import imgkit

def capturar_pagina(url, caminho_salvar):
    # Configura o caminho do wkhtmltoimage (caso necessário)
    config = imgkit.config(wkhtmltoimage='C:/caminho/para/wkhtmltoimage.exe')
    
    # Captura a página como uma imagem e salva no caminho especificado
    imgkit.from_url(url, caminho_salvar, config=config)
    print(f"Página capturada e salva em: {caminho_salvar}")

# URL da página que você quer capturar
url = "https://exemplo.com"
