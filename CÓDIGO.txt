import os

def agrupar_arquivos(diretorio_entrada, caminho_saida):
    with open(caminho_saida, 'w') as arquivo_final:
        for nome_arquivo in os.listdir(diretorio_entrada):
            if nome_arquivo.endswith('.txt'):
                caminho = os.path.join(diretorio_entrada, nome_arquivo)
                with open(caminho, 'r') as f:
                    conteudo = f.read()
                    arquivo_final.write(conteudo)
                    arquivo_final.write("\n")  # Adiciona uma nova linha entre os conteúdos dos arquivos

diretorio_entrada = '/caminho/para/diretorio'
caminho_saida = '/caminho/para/arquivo_final.txt'

# Verifica se o diretório de entrada existe
if not os.path.isdir(diretorio_entrada):
    print(f"Diretório não encontrado: {diretorio_entrada}")
    exit()

agrupar_arquivos(diretorio_entrada, caminho_saida)
print(f"Arquivos foram agrupados em {caminho_saida}")
