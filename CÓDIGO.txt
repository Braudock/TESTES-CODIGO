import webbrowser
import pyautogui
import os
import time

# URL a ser aberta
url = "https://sua-url-aqui.com"
webbrowser.open(url)

# Aguardar alguns segundos para o navegador abrir a URL completamente
time.sleep(5)  # Ajuste conforme o tempo necessário para o carregamento

# Coordenadas e o código a ser inserido
codigo = "SEU_CODIGO_AQUI"

# Etapa 1: Ir para (x=677, y=437) e clicar
pyautogui.click(x=677, y=437)

# Inserir o código após o clique
pyautogui.typewrite(codigo)

# Etapa 2: Ir para (x=809, y=475) e clicar
pyautogui.click(x=809, y=475)

# Etapa 3: Capturar a área específica da tela
# Caminho da pasta onde deseja salvar a imagem
pasta_destino = r"C:\Users\BRAUD\OneDrive\Área de Trabalho\PASTAS"
nome_arquivo = "imagem_capturada.jpg"
caminho_completo = os.path.join(pasta_destino, nome_arquivo)

# Coordenadas e tamanho da área da imagem
left, top, width, height = 85, 337, 98, 124
screenshot = pyautogui.screenshot(region=(left, top, width, height))

# Salva a captura na pasta especificada
screenshot.save(caminho_completo)
print(f"Imagem capturada e salva em: {caminho_completo}")
