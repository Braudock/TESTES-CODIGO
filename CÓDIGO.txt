from pdf2docx import parse

# Definindo o arquivo PDF de entrada
arquivo_pdf = r"C:\Users\BRAUD\PROPOSTA.pdf"

# Definindo o arquivo Word de saída
arquivo_docx = r"C:\Users\BRAUD\meu_arquivo.docx"

# Convertendo o PDF para DOCX

print(f"Conversão de {arquivo_pdf} para {arquivo_docx} concluída!")
