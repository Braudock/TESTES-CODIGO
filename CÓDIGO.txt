PROC SQL;
   CREATE TABLE VP_TOTAL AS
   SELECT DISTINCT t1.*
   FROM VP_TOTAL_1 AS t1
   RIGHT JOIN &MES_ATUAL..METAS_&MES_ATUAL AS t2
      ON COMPRESS(t1.SEGMENTO) = t2.SEGMENTO
         AND t1.COD_PLATAFORMA = t2.PLATAFORMA
         AND t1.GERENTE = SUBSTR(PUT(t2.CODASSIST, BEST12.), 1, 3);
QUIT;
