import requests
from bs4 import BeautifulSoup

# URL de exemplo (pode substituir por qualquer URL válida)
url = "https://www.example.com"

# Fazendo uma requisição HTTP para obter o conteúdo da página
response = requests.get(url)

# Verificando se a requisição foi bem-sucedida
if response.status_code == 200:
    # Usando BeautifulSoup para analisar o HTML
    soup = BeautifulSoup(response.text, 'html.parser')

    # Extraindo o título da página
    titulo = soup.title.string
    print("Título da página:", titulo)

    # Extraindo todos os links (href) na página
    for link in soup.find_all('a', href=True):
        print("Link encontrado:", link['href'])

else:
    print("Erro ao acessar a página:", response.status_code)
