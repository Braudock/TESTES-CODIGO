import requests
from bs4 import BeautifulSoup

# URL do site que você quer acessar
url = 'https://example.com'  # Substitua pelo URL desejado

# Fazendo uma requisição GET
response = requests.get(url)

# Verificando se a requisição foi bem-sucedida
if response.status_code == 200:
    # Analisando o conteúdo HTML da página
    soup = BeautifulSoup(response.content, 'html.parser')

    # Exemplo: Encontrando todos os links na página
    links = soup.find_all('a')
    for link in links:
        print(link.get('href'))

    # Exemplo: Enviando um formulário
    form_data = {
        'campo1': 'valor1',
        'campo2': 'valor2',
        # Adicione mais campos conforme necessário
    }
    form_action = soup.find('form').get('action')  # Obtendo a ação do formulário
    if not form_action.startswith('http'):
        form_action = url + form_action  # Ajustando a URL se necessário

    # Enviando a requisição POST
    post_response = requests.post(form_action, data=form_data)
    if post_response.status_code == 200:
        print('Formulário enviado com sucesso!')
        # Faça algo com a resposta do formulário, se necessário
else:
    print('Erro ao acessar o site:', response.status_code)
