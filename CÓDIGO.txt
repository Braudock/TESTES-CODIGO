import pyautogui
import keyboard

print("Posicione o mouse e pressione a tecla 'p' para capturar as coordenadas. Pressione 'q' para sair.")

while True:
    if keyboard.is_pressed('p'):  # Verifica se a tecla 'p' foi pressionada
        x, y = pyautogui.position()
        print(f"Coordenadas do mouse: x={x}, y={y}")
        
        # Aguarda até que a tecla 'p' seja liberada para evitar múltiplas capturas
        while keyboard.is_pressed('p'):
            pass
    
    # Finaliza o programa quando a tecla 'q' é pressionada
    if keyboard.is_pressed('q'):
        print("Programa finalizado.")
        break
