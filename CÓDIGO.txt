import pyautogui
import time

# Certifique-se de que o Teams esteja aberto e na janela correta.
# Simula as teclas para selecionar tudo (Ctrl+A), copiar (Ctrl+C) e acessar o conteúdo copiado.

# Pausa para ajustar a janela (5 segundos, por exemplo)
time.sleep(5)

# Simula Ctrl+A para selecionar todas as mensagens da janela
pyautogui.hotkey('ctrl', 'a')

# Simula Ctrl+C para copiar o conteúdo selecionado
pyautogui.hotkey('ctrl', 'c')

# Agora o conteúdo está na área de transferência
import pyperclip

# Obtém o texto da área de transferência
conversas = pyperclip.paste()

# Exibe as conversas copiadas
print(conversas)
