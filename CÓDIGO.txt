import pyautogui
import webbrowser
import time

# Passo 1: Abrir a URL
url = "https://exemplo.com"  # Insira a URL desejada aqui
webbrowser.open(url)

# Aguardar a página carregar
time.sleep(5)  # Ajuste o tempo de acordo com a velocidade de carregamento da página

# Passo 2: Ir para a coordenada X=466, Y=295 e clicar
pyautogui.click(466, 295)
time.sleep(1)

# Escrever o código ""
pyautogui.write("")
time.sleep(1)

# Passo 3: Ir para a coordenada X=677, Y=437, clicar com o botão direito e selecionar "Salvar imagem como"
pyautogui.rightClick(677, 437)
time.sleep(1)

# Navegar no menu para selecionar "Salvar imagem como"
# Pressione 'down' algumas vezes até chegar na opção "Salvar imagem como"
pyautogui.press('down')  # Ajuste para a posição correta da opção no menu
time.sleep(1)

# Confirmar a seleção com Enter
pyautogui.press('enter')
time.sleep(1)

# Passo 4: Ir para a coordenada X=809, Y=474 e clicar com o botão direito para selecionar "Salvar como"
pyautogui.rightClick(809, 474)
time.sleep(1)

# Pressionar 'down' para selecionar "Salvar como"
pyautogui.press('down')
time.sleep(1)

# Confirmar a seleção com Enter
pyautogui.press('enter')
time.sleep(1)

# Passo 5: Ir para a coordenada X=160, Y=398, clicar e escrever "IMAGEM"
pyautogui.click(160, 398)
time.sleep(1)
pyautogui.write("IMAGEM")
time.sleep(1)

# Passo 6: Ir para a coordenada X=160, Y=501 e clicar
pyautogui.click(160, 501)
time.sleep(1)

# Passo 7: Ir para a coordenada X=782, Y=398 e finalizar
pyautogui.click(782, 398)

print("A automação foi concluída.")
