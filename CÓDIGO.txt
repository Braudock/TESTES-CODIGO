import asyncio
from pyppeteer import launch

async def preencher_campo():
    # Inicia o navegador
    browser = await launch(headless=False)  # headless=False permite ver a ação no navegador
    page = await browser.newPage()

    # Acessa a página desejada
    await page.goto('')

    # Aguarda a página carregar completamente
    await page.waitForSelector('input[placeholder="Procure um Colaborador"]')

    # Insere o texto no campo de busca
    await page.type('input[placeholder="Procure um Colaborador"]', 'nome do colaborador')

    # Opcional: Fecha o navegador após a execução
    # await browser.close()

# Executa a função assíncrona
asyncio.get_event_loop().run_until_complete(preencher_campo())
