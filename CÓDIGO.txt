import pyautogui
import keyboard

print("Posicione o cursor do mouse sobre o canto superior esquerdo da área e pressione 'Enter' para capturar.")

# Aguardando o usuário pressionar 'Enter' para capturar o canto superior esquerdo
keyboard.wait('enter')
x1, y1 = pyautogui.position()
print(f"Canto superior esquerdo capturado em: ({x1}, {y1})")

print("Agora, posicione o cursor do mouse sobre o canto inferior direito da área e pressione 'Enter' para capturar.")

# Aguardando o usuário pressionar 'Enter' para capturar o canto inferior direito
keyboard.wait('enter')
x2, y2 = pyautogui.position()
print(f"Canto inferior direito capturado em: ({x2}, {y2})")

# Calcula a largura e a altura da área
width = x2 - x1
height = y2 - y1

print(f"Coordenadas da área para captura: left={x1}, top={y1}, width={width}, height={height}")
