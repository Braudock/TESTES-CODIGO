Sub AbrirChromeComURL()
    On Error Resume Next

    Dim ShellApp As Object
    Dim URL As String
    Dim Codigo As String

    ' URL e código a ser inserido
    URL = ""
    Codigo = "123456"  ' Substitua pelo código que você deseja inserir

    ' Abrir Chrome com URL
    Set ShellApp = CreateObject("Shell.Application")
    ShellApp.ShellExecute "chrome.exe", URL, "", "open", 1

    ' Verificar erro ao abrir o Chrome
    If Err.Number <> 0 Then
        MsgBox "Ocorreu um erro ao abrir o Chrome: " & Err.Description
        Exit Sub
    End If

    ' Aguardar alguns segundos para a página carregar
    Application.Wait Now + TimeValue("0:00:05")

    ' Usar SendKeys para inserir o código
    SendKeys Codigo, True
    SendKeys "{TAB}", True  ' Opcional: usar TAB para navegar entre campos, se necessário
    SendKeys "{ENTER}", True ' Opcional: pressionar Enter para submeter o campo, se aplicável

    ' Limpar variáveis
    Set ShellApp = Nothing
End Sub
