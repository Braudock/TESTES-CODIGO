import pyautogui
import time

print("Posicione o cursor do mouse sobre o canto superior esquerdo da 치rea que deseja capturar.")
time.sleep(5)  # Tempo para posicionar o cursor

# Captura as coordenadas do canto superior esquerdo
x1, y1 = pyautogui.position()
print(f"Canto superior esquerdo: ({x1}, {y1})")

print("Agora, posicione o cursor do mouse sobre o canto inferior direito da 치rea que deseja capturar.")
time.sleep(5)  # Tempo para posicionar o cursor

# Captura as coordenadas do canto inferior direito
x2, y2 = pyautogui.position()
print(f"Canto inferior direito: ({x2}, {y2})")

# Calcula a largura e a altura da 치rea
width = x2 - x1
height = y2 - y1

print(f"Coordenadas da 치rea para captura: left={x1}, top={y1}, width={width}, height={height}")
