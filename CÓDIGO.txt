import asyncio
from pyppeteer import launch

async def acessar_pagina():
    browser = await launch()
    page = await browser.newPage()
    await page.goto("http://url-do-site.com")
    # Adicione aqui os passos para interagir com a p√°gina
    await browser.close()

asyncio.get_event_loop().run_until_complete(acessar_pagina())
